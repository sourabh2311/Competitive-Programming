    The problem of finding discrete root is defined as follows. Given a prime nn and two integers aa and kk, find all xx for which:

    xk≡a(modn)xk≡a(modn)

    The algorithm

    We will solve this problem by reducing it to the discrete logarithm problem.

    Let's apply the concept of a primitive root modulo nn. Let gg be a primitive root modulo nn. Note that since nn is prime, it must exist, and it can be found in O(Ans⋅logϕ(n)⋅logn)=O(Ans⋅log2n)O(Ans⋅log⁡ϕ(n)⋅log⁡n)=O(Ans⋅log2⁡n) plus time of factoring ϕ(n)ϕ(n).

    We can easily discard the case where a=0a=0. In this case, obviously there is only one answer: x=0x=0.

    Since we jnow that nn is a prime, any number between 1 and n−1n−1 can be represented as a power of the primitive root, and we can represent the discrete root problem as follows:

    (gy)k≡a(modn)(gy)k≡a(modn)

    where

    x≡gy(modn)x≡gy(modn)

    This, in turn, can be rewritten as

    (gk)y≡a(modn)(gk)y≡a(modn)

    Now we have one unknown yy, which is a discrete logarithm problem. The solution can be found using Shanks' baby-step-giant-step algorithm in O(n−−√logn)O(nlog⁡n) (or we can verify that there are no solutions).

    Having found one solution y0y0, one of solutions of discrete root problem will be x0=gy0(modn)x0=gy0(modn).

    Finding all solutions from one known solution

    To solve the given problem in full, we need to find all solutions knowing one of them: x0=gy0(modn)x0=gy0(modn).

    Let's recall the fact that a primitive root always has order of ϕ(n)ϕ(n), i.e. the smallest power of gg which gives 1 is ϕ(n)ϕ(n). Therefore, if we add the term ϕ(n)ϕ(n) to the exponential, we still get the same value:

    xk≡gy0⋅k+l⋅ϕ(n)≡a(modn)∀l∈Zxk≡gy0⋅k+l⋅ϕ(n)≡a(modn)∀l∈Z

    Hence, all the solutions are of the form:

    x=gy0+l⋅ϕ(n)k(modn)∀l∈Zx=gy0+l⋅ϕ(n)k(modn)∀l∈Z.

    where ll is chosen such that the fraction must be an integer. For this to be true, the numerator has to be divisible by the least common multiple of ϕ(n)ϕ(n) and kk. Remember that least common multiple of two numbers lcm(a,b)=a⋅bgcd(a,b)lcm(a,b)=a⋅bgcd(a,b); we'll get

    x=gy0+iϕ(n)gcd(k,ϕ(n))(modn)∀i∈Zx=gy0+iϕ(n)gcd(k,ϕ(n))(modn)∀i∈Z.

    This is the final formula for all solutions of discrete root problem.

    Implementation

    Here is a full implementation, including routines for finding the primitive root, discrete log and finding and printing all solutions.

    int gcd (int a, int b) {
    return a ? gcd (b%a, a) : b;
}

int powmod (int a, int b, int p) {
    int res = 1;
    while (b)
        if (b & 1)
            res = int (res * 1ll * a % p),  --b;
        else
            a = int (a * 1ll * a % p),  b >>= 1;
    return res;
}

int generator (int p) {
    vector<int> fact;
    int phi = p-1,  n = phi;
    for (int i=2; i*i<=n; ++i)
        if (n % i == 0) {
            fact.push_back (i);
            while (n % i == 0)
                n /= i;
        }
    if (n > 1)
        fact.push_back (n);

    for (int res=2; res<=p; ++res) {
        bool ok = true;
        for (size_t i=0; i<fact.size() && ok; ++i)
            ok &= powmod (res, phi / fact[i], p) != 1;
        if (ok)  return res;
    }
    return -1;
}

int main() {

    int n, k, a;
    cin >> n >> k >> a;
    if (a == 0) {
        puts ("1\n0");
        return 0;
    }

    int g = generator (n);

    int sq = (int) sqrt (n + .0) + 1;
    vector < pair<int,int> > dec (sq);
    for (int i=1; i<=sq; ++i)
        dec[i-1] = make_pair (powmod (g, int (i * sq * 1ll * k % (n - 1)), n), i);
    sort (dec.begin(), dec.end());
    int any_ans = -1;
    for (int i=0; i<sq; ++i) {
        int my = int (powmod (g, int (i * 1ll * k % (n - 1)), n) * 1ll * a % n);
        vector < pair<int,int> >::iterator it =
            lower_bound (dec.begin(), dec.end(), make_pair (my, 0));
        if (it != dec.end() && it->first == my) {
            any_ans = it->second * sq - i;
            break;
        }
    }
    if (any_ans == -1) {
        puts ("0");
        return 0;
    }

    int delta = (n-1) / gcd (k, n-1);
    vector<int> ans;
    for (int cur=any_ans%delta; cur<n-1; cur+=delta)
        ans.push_back (powmod (g, cur, n));
    sort (ans.begin(), ans.end());
    printf ("%d\n", ans.size());
    for (size_t i=0; i<ans.size(); ++i)
        printf ("%d ", ans[i]);

}
